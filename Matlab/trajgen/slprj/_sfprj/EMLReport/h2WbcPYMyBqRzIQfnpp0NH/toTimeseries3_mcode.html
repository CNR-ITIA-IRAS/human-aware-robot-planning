1function [ tss, st, et ] = toTimeseries( opennitracker_data )
 2
 3persistent frames_used;
 4if isempty( frames_used )
 5    frames_used =[ '/skel/SpineBase' ...
 6                 , '/skel/SpineMid'  ...
 7                 , '/skel/Neck' ...
 8                 , '/skel/Head' ...
 9                 , '/skel/ShoulderLeft' ...
10                 , '/skel/ElbowLeft' ...
11                 , '/skel/WristLeft' ...
12                 , '/skel/HandLeft' ...
13                 , '/skel/ShoulderRight' ...
14                 , '/skel/ElbowRight' ...
15                 , '/skel/WristRight' ...
16                 , '/skel/HandRight' ...
17                 , '/skel/HipLeft' ...
18                 , '/skel/KneeLeft' ...
19                 , '/skel/AnkleLeft' ...
20                 , '/skel/FootLeft' ...
21                 , '/skel/HipRight' ...
22                 , '/skel/KneeRight' ...
23                 , '/skel/AnkleRight' ...
24                 , '/skel/FootRight' ...
25                 , '/skel/SpineShoulder' ...
26                 , '/skel/HandTipLeft' ...
27                 , '/skel/ThumbLeft' ...
28                 , '/skel/HandTipRight' ...
29                 , '/skel/ThumbRight' ];
30end
31
32tss   = cell( length( frames_used ), 1 );
33tdd   = cell( length( frames_used ), 2 );
34
35for t = 1:length( opennitracker_data )
36    % t , { '/skel/ElbowLeft', '/skel/ElbowRight', '/skel/HandLeft', '/skel/HandRight', '/skel/Head', '/skel/HipLeft', '/skel/HipRight',}, 
37    % [ [ 0.231377, -0.799918, 0.168197, ]; [ 0.127866, -1.12947, 0.198706, ]; [ 0.433153, -0.85764, 0.154607, ]; [ 0.404257, -0.938758, 0.161978, ]; [ 0.158786, -0.853775, 0.700064, ]; [ 0.174011, -0.841702, -0.0278721, ]; [ 0.156605, -0.993377, -0.00360158, ]; [ 0.149648, -0.918245, 0.588661, ]; [ 0.186534, -0.792811, 0.433465, ]; [ 0.107367, -1.02325, 0.421039, ]; [ 0.383672, -0.773916, 0.138229, ]; [ 0.334761, -0.982525, 0.147994, ]; ] }
38    opennitracker = opennitracker_data{t};
39    frames_names_   = opennitracker{2};
40    frames_data_    = opennitracker{3};
41    for iFrame = 1:length( frames_names_  )
42       idx            = find( frames_names_ == frames_used( iFrame ) );
43       if isempty( idx )
44            continue;
45       else
46            tdd{iFrame}{1} = opennitracker{1};
47            tdd{iFrame}{2} = [ tss{iFrame}; frames_data_{ idx, : } ];
48       end
49    end
50end
51
52st = 0;
53et = opennitracker_data{end}{1}-opennitracker_data{1}{1};
54for iTss = 1:length( frames_used )
55    tss{iTss} = spline  ( tdd{iFrame}{1} ...
56                        , tdd{iFrame}{2} ...
57                        , 0.0:0.25:opennitracker_data{end}{1}-opennitracker_data{1}{1} );
58end
